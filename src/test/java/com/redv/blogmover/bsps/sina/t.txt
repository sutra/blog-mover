function AddButtonAction(action) {
	var _cfg = {name: "SuperTable0", cgi: action};
	var sFormName = "_form_" + _cfg["name"] + "_";
	var sFrameName = "_frame_" + _cfg["name"] + "_";
	var oFormName = document.getElementById(sFormName);
	var oFrameName = document.getElementById(sFrameName);
	if(oFormName != null) {
		oFormName.removeNode(true);
	}
	if(oFrameName != null) {
		oFrameName.removeNode(true);
	}
	if(window.Event) {
		var oFrame = document.createElement("iframe");
		oFrame.name = sFrameName;
	}
	else {
		var oFrame = document.createElement("<iframe name='" + sFrameName + "'></iframe>");
	}
	oFrame.id = sFrameName;
	oFrame.style.display = "none";
	var oForm = document.createElement("form");
	oForm.id = sFormName;
	oForm.name = sFormName;
	oForm.action = _cfg["cgi"];
	oForm.method = "post";
	oForm.style.display = "none";
	oForm.target = sFrameName;
	var k = 0;
	for(var i = 0; i < arr1.length; i++) {
		if(arr1[i]["cfg_tr_click"]) {
			oInput = document.createElement("input");
			//oInput.type = "hidden";
			oInput.value = arr1[i]["data"][1];
			oInput.name = arr0["form"][1]+"[]";
			oForm.appendChild(oInput);

			oInput = document.createElement("input");
			//oInput.type = "hidden";
			oInput.value = arr1[i]["hidden"][0];
			oInput.name = arr0["hidden"][0]+"[]";
			oForm.appendChild(oInput);
			oForm.appendChild(document.createElement("br"));
			k ++;
		}
	}
	if(k == 0)return;
	document.body.appendChild(oForm);
	document.body.appendChild(oFrame);
	oForm.submit();
}
